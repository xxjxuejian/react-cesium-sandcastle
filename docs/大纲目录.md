# 项目大纲目录

以下目录结构基于当前仓库整理，已省略 `node_modules/`、`dist/`、`.git/` 等依赖、构建产物和版本控制目录。

```text
react-cesium-sandcastle/
├── AGENTS.md                       # 仓库贡献者与智能体协作指南
├── README.md                       # 项目基础说明与 Vite 模板参考
├── package.json                    # 项目元信息、依赖与脚本命令
├── pnpm-lock.yaml                  # pnpm 依赖锁定文件
├── index.html                      # Vite 应用 HTML 入口
├── vite.config.ts                  # Vite 构建与开发服务配置
├── tsconfig.json                   # TypeScript 配置入口
├── tsconfig.app.json               # 应用源码 TypeScript 配置
├── tsconfig.node.json              # Node 侧配置文件 TypeScript 配置
├── eslint.config.js                # ESLint 规则配置
├── tailwind.config.js              # Tailwind CSS 配置
├── postcss.config.js               # PostCSS 配置
├── .prettierrc                     # Prettier 插件配置
├── .gitignore                      # Git 忽略规则
├── .cursor/                        # Cursor 编辑器相关配置
├── .vscode/                        # VS Code 工作区配置
├── docs/                           # 项目文档目录
│   ├── i18n配置和使用.md           # 国际化配置与使用说明
│   ├── 目录结构.md                 # 已有目录结构说明文档
│   └── 大纲目录.md                 # 当前文件，描述项目目录大纲
├── public/                         # 直接作为静态资源发布的文件
│   ├── favicon.svg                 # 浏览器标签页图标
│   └── icons.svg                   # 公共图标资源
└── src/                            # 应用源码目录
    ├── main.tsx                    # React 应用挂载入口
    ├── index.css                   # 全局基础样式
    ├── app/                        # 应用根组件与全局 Provider
    │   ├── App.tsx                 # 应用顶层组件
    │   └── providers/              # 全局上下文与能力注入
    │       ├── AppProvider.tsx     # 应用级 Provider 聚合
    │       ├── I18nProvider.tsx    # 国际化 Provider
    │       └── ThemeProvider.tsx   # 主题 Provider 预留文件
    ├── assets/                     # 源码内引用的图片与 SVG 资源
    │   ├── hero.png                # 首页或展示用图片资源
    │   ├── react.svg               # React 标识资源
    │   └── vite.svg                # Vite 标识资源
    ├── components/                 # 可复用业务或 UI 组件
    │   └── LanguageSwitch.tsx      # 语言切换组件
    ├── config/                     # 静态配置
    │   └── menu.ts                 # 菜单配置数据
    ├── hooks/                      # 自定义 React Hooks 目录
    ├── i18n/                       # 国际化初始化与语言资源
    │   ├── index.ts                # i18n 初始化入口
    │   └── locales/                # 多语言文案
    │       ├── en-US.ts            # 英文文案
    │       └── zh-CN.ts            # 中文文案
    ├── layouts/                    # 页面布局组件
    │   ├── index.tsx               # 布局导出入口
    │   └── MainLayout/             # 主布局模块
    │       ├── index.tsx           # 主布局容器
    │       ├── Header.tsx          # 顶部栏
    │       ├── Sidebar.tsx         # 侧边导航
    │       ├── MobileDrawer.tsx    # 移动端抽屉导航
    │       └── Content.tsx         # 内容区域容器
    ├── pages/                      # 路由页面目录
    │   ├── index.md                # 页面说明或索引文档
    │   ├── home/                   # 首页页面
    │   │   └── index.tsx           # 首页组件
    │   ├── error/                  # 错误页面
    │   │   └── NotFound.tsx        # 404 页面组件
    │   ├── getting-started/        # 入门示例页面
    │   │   ├── hello-world/        # Hello World 示例
    │   │   │   └── index.tsx       # 示例页面组件
    │   │   └── resolution-scale/   # 分辨率缩放示例
    │   │       └── index.tsx       # 示例页面组件
    │   ├── animation/              # 动画相关示例页面
    │   │   └── clock/              # 时钟动画示例
    │   │       └── index.tsx       # 示例页面组件
    │   └── showcases/              # 综合展示示例页面
    │       └── google-2d-tiles/    # Google 2D 瓦片示例
    │           └── index.tsx       # 示例页面组件
    ├── router/                     # 路由声明、转换与菜单映射
    │   ├── index.tsx               # 路由入口
    │   ├── menu.tsx                # 路由菜单生成逻辑
    │   ├── mockRoutes.ts           # 示例或模拟路由数据
    │   ├── routeModules.ts         # 页面模块收集配置
    │   ├── transform.tsx           # 路由数据转换逻辑
    │   └── types.ts                # 路由相关类型定义
    ├── styles/                     # 全局或主题样式
    │   └── theme.css               # 主题样式预留文件
    └── types/                      # 全局类型声明目录
```
